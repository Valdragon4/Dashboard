## Story

**As a** utilisateur,
**I want** pouvoir connecter mes comptes bancaires avec stockage sécurisé des credentials,
**so that** le système peut synchroniser automatiquement mes données.

from django.test import TestCase
from django.contrib.auth import get_user_model
from finance.models import Account, BankConnection, SyncLog
from finance.services.encryption_service import EncryptionService

User = get_user_model()

class TestBankConnection(TestCase):
    def setUp(self):
        self.user = User.objects.create_user(username='test', password='test')
        self.account = Account.objects.create(
            owner=self.user,
            name="Test Account",
            type="checking"
        )
    
    def test_create_bank_connection_with_encrypted_credentials(self):
        """Test création de BankConnection avec credentials chiffrés."""
        credentials = {"username": "test", "password": "secret"}
        encrypted = EncryptionService.encrypt_credentials(credentials)
        
        connection = BankConnection.objects.create(
            owner=self.user,
            provider="trade_republic",
            account_name="Test Account",
            encrypted_credentials=encrypted,
            auto_sync_enabled=True
        )
        
        self.assertIsNotNone(connection.id)
        self.assertEqual(connection.provider, "trade_republic")
        
        # Vérifier déchiffrement
        decrypted = EncryptionService.decrypt_credentials(connection.encrypted_credentials)
        self.assertEqual(decrypted["username"], "test")
